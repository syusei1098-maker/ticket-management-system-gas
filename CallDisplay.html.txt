  <script>
    function load() {
      google.script.run.withSuccessHandler(render).getCallList();
    }

    function render(list) {
      const container = document.getElementById('list-container');
      container.innerHTML = '';
      
      if (list.length === 0) {
        container.innerHTML = '<div class="empty-msg">現在呼び出し中の番号はありません</div>';
        return;
      }
      
      list.forEach(t => {
        const div = document.createElement('div');
        div.className = 'ticket-card';
        div.innerHTML = `
          <span class="status-label">出来上がり</span>
          <div class="ticket-no">${t.no}</div>
          <div class="customer-name">${t.name} 様</div>
          <div class="items">${t.items}</div>
        `;
        container.appendChild(div);
      });
    }

    // 初回実行と定期実行
    load();
    setInterval(load, 15000); 
  </script>
