    <script>
      // 初期ロード時の設定
      toggleEmailReq();

      function toggleEmailReq() {
        const method = document.getElementById('method').value;
        const emailInput = document.getElementById('email');
        const reqMark = document.getElementById('email-req-mark');
        
        if (method === 'email') {
          emailInput.required = true;
          reqMark.classList.remove('hidden');
        } else {
          emailInput.required = false;
          reqMark.classList.add('hidden');
        }
      }

      function order(e) {
        e.preventDefault();
        const btn = document.getElementById('btn');
        btn.disabled = true;
        btn.innerText = '処理中...';

        var orders = [];
        document.querySelectorAll('.qty').forEach(el => {
          if(el.value > 0) orders.push({ item: el.dataset.item, qty: el.value });
        });
        
        if(orders.length === 0) {
           btn.disabled = false;
           btn.innerText = 'チケットを発行する';
           return alert('商品を1つ以上選択してください');
        }

        var data = {
          name: document.getElementById('name').value,
          email: document.getElementById('email').value,
          phone: document.getElementById('phone').value,
          method: document.getElementById('method').value,
          orders: orders
        };

        google.script.run.withSuccessHandler(done).processTicketIssue(data);
      }

      function done(res) {
        if(!res.success) {
          alert(res.message);
          const btn = document.getElementById('btn');
          btn.disabled = false;
          btn.innerText = 'チケットを発行する';
          return;
        }
        
        document.getElementById('order-view').classList.add('hidden');
        document.getElementById('result-view').classList.remove('hidden');
        document.getElementById('res-no').innerText = res.ticketNo;
        document.getElementById('res-cost').innerText = res.totalCost.toLocaleString();
        document.getElementById('msg').innerText = res.message;
        
        const summaryList = document.getElementById('summary-list');
        summaryList.innerHTML = '';
        Object.keys(res.summary).forEach(item => {
          const li = document.createElement('li');
          li.innerText = `${item} × ${res.summary[item].qty}`;
          summaryList.appendChild(li);
        });

        if(res.redeemUrl) {
          document.getElementById("qr-area").innerHTML = '';
          new QRCode(document.getElementById("qr-area"), { text: res.redeemUrl, width: 180, height: 180 });
        }
      }
