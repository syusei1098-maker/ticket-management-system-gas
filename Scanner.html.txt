        <div class="manual-input-flex">
          <input type="text" id="manual-ticket-no" placeholder="例: 5 または No.005">
          <button id="manual-btn" onclick="submitManual()">受渡</button>
        </div>

  <script>
    let CACHED_PASS = '';
    let html5QrcodeScanner = null;
    let currentVideoTrack = null;

    function login() {
      const passInput = document.getElementById('pass').value;
      if (!passInput) return alert('パスワードを入力してください');

      google.script.run.withSuccessHandler(res => {
        if (res.success) {
          CACHED_PASS = passInput;
          document.getElementById('login-view').classList.add('hidden');
          startScanner();
        } else {
          alert('パスワードが違います');
          document.getElementById('pass').value = '';
        }
      }).checkStaffLogin(passInput);
    }

    function startScanner() {
      document.getElementById('scanner-view').classList.remove('hidden');
      document.getElementById('result-view').classList.add('hidden');
      document.getElementById('manual-ticket-no').value = ''; // 入力欄リセット

      if (!html5QrcodeScanner) {
        html5QrcodeScanner = new Html5Qrcode("reader");
      }

      const config = { 
        fps: 10, 
        qrbox: { width: 250, height: 250 },
        videoConstraints: {
           facingMode: "environment",
           width: { min: 640, ideal: 1280, max: 1920 },
           height: { min: 480, ideal: 720, max: 1080 }
        }
      };

      html5QrcodeScanner.start(
        { facingMode: "environment" }, 
        config,
        onScanSuccess,
        onScanFailure
      ).then(() => {
        setupZoom();
      }).catch(err => {
        console.error(err);
        // カメラ起動失敗時も手動入力は使えるようにする
      });
    }

    function setupZoom() {
      const zoomArea = document.getElementById('zoom-area');
      const slider = document.getElementById('zoom-slider');
      
      if (html5QrcodeScanner) {
         const videoElement = document.querySelector("#reader video");
         if (videoElement && videoElement.srcObject) {
            const track = videoElement.srcObject.getVideoTracks()[0];
            const capabilities = track.getCapabilities();

            if (capabilities.zoom) {
               currentVideoTrack = track;
               zoomArea.style.display = 'block';
               slider.min = capabilities.zoom.min;
               slider.max = capabilities.zoom.max;
               slider.step = capabilities.zoom.step || 0.1;
               slider.value = track.getSettings().zoom || 1;
            }
         }
      }
    }

    function applyZoom(val) {
      if (currentVideoTrack) {
        currentVideoTrack.applyConstraints({ advanced: [{ zoom: parseFloat(val) }] });
      }
    }

    // --- ① QRコードスキャン成功時 ---
    function onScanSuccess(decodedText, decodedResult) {
      if(html5QrcodeScanner) html5QrcodeScanner.pause();
      let token = decodedText;
      if(decodedText.includes('ticket=')) {
        token = decodedText.split('ticket=')[1];
      }
      google.script.run.withSuccessHandler(showResult).processQrScan(token, CACHED_PASS);
    }

    function onScanFailure(error) { /* 無視 */ }

    // --- ② 手動入力ボタン押下時 ---
    function submitManual() {
      const inputNo = document.getElementById('manual-ticket-no').value;
      if(!inputNo) return alert('整理番号を入力してください');

      const btn = document.getElementById('manual-btn');
      btn.disabled = true;
      btn.innerText = '処理中';

      // 処理中にQRが誤反応しないよう一時停止
      if(html5QrcodeScanner) {
         try { html5QrcodeScanner.pause(); } catch(e){}
      }

      // 新しく追加した手動入力用関数を呼び出す
      google.script.run.withSuccessHandler(res => {
         btn.disabled = false;
         btn.innerText = '受渡';
         showResult(res);
      }).processManualRedemption(inputNo, CACHED_PASS);
    }

    // --- 結果表示 ---
    function showResult(res) {
      document.getElementById('scanner-view').classList.add('hidden');
      document.getElementById('result-view').classList.remove('hidden');
      
      if(html5QrcodeScanner) {
        html5QrcodeScanner.stop().then(() => { html5QrcodeScanner.clear(); }).catch(e=>{});
      }

      const resBox = document.querySelector('.result-box');
      if (res.success) {
        resBox.style.borderColor = "var(--main-red)";
        document.querySelector('.success-msg').innerText = "受渡完了";
        document.querySelector('.success-msg').style.color = "#2e7d32";
        document.getElementById('res-ticket').innerText = res.ticketNo;
        document.getElementById('res-name').innerText = res.name + " 様";
      } else {
        resBox.style.borderColor = "#555";
        document.querySelector('.success-msg').innerText = "エラー";
        document.querySelector('.success-msg').style.color = "#c62828";
        document.getElementById('res-ticket').innerText = "";
        document.getElementById('res-name').innerText = res.message;
      }
    }

    function resetScanner() {
      html5QrcodeScanner = null;
      startScanner();
    }
    
    function stopScan() {
        if(html5QrcodeScanner) {
             html5QrcodeScanner.stop().then(() => { html5QrcodeScanner.clear(); }).catch(e=>{});
        }
        document.getElementById('scanner-view').classList.add('hidden');
        document.getElementById('login-view').classList.remove('hidden');
        CACHED_PASS = ''; 
    }
  </script>
