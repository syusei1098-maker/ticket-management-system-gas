  <script>
    let PASS = '';

    function login() {
      PASS = document.getElementById('pass').value;
      if (!PASS) return alert('パスワードを入力してください');
      google.script.run.withSuccessHandler(res => {
        if (res.success) {
          document.getElementById('auth-box').style.display = 'none';
          document.getElementById('kitchen-view').style.display = 'block';
          loadData();
        } else {
          alert('パスワードが違います');
          document.getElementById('pass').value = '';
        }
      }).checkStaffLogin(PASS);
    }

    function loadData() {
      google.script.run.withSuccessHandler(renderList).getKitchenStatus();
    }

    function renderList(items) {
      const container = document.getElementById('order-list');
      container.innerHTML = '';
      
      items.forEach(row => {
        const div = document.createElement('div');
        // 待ちがある場合は赤枠、なければ黒枠
        div.className = row.stats.waiting > 0 ? 'card has-wait' : 'card';
        const inputId = `num-${row.item.replace(/\s/g, '-')}`; 

        div.innerHTML = `
          <div class="info-row">
            <h3 class="item-name">${row.item}</h3>
            <div style="text-align:right;">
              <span class="status-badge status-wait">待: ${row.stats.waiting}</span>
              <span class="status-badge status-stock">在庫: ${row.stats.buffer}</span>
            </div>
          </div>
          <div class="controls">
            <button class="btn-add" onclick="addCount('${inputId}', 1)">+1</button>
            <button class="btn-add" onclick="addCount('${inputId}', 5)">+5</button>
            <input type="number" id="${inputId}" class="num-input" placeholder="0">
            <button class="btn-complete" onclick="complete('${row.item}', '${inputId}')">完了登録</button>
          </div>
        `;
        container.appendChild(div);
      });
    }

    function addCount(id, n) {
      const el = document.getElementById(id);
      let val = parseInt(el.value) || 0;
      el.value = val + n;
    }

    function complete(item, id) {
      const input = document.getElementById(id);
      const count = parseInt(input.value);
      if(!count || count <= 0) return alert('数量を入力してください');
      
      if(!confirm(`${item} を ${count}個 作成完了しますか？\n(待ちが無い場合は在庫になります)`)) return;
      
      google.script.run.withSuccessHandler(res => {
        alert(res.message);
        loadData();
        input.value = 0;
      }).registerCookedItems(item, count, PASS);
    }
    
    setInterval(() => { if(PASS) loadData(); }, 15000);
  </script>
